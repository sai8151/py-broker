CREATE TABLE users (
	id INTEGER NOT NULL, 
	seller_id VARCHAR, 
	password VARCHAR, 
	seller_name VARCHAR, 
	bank_name_1 VARCHAR, 
	bank_account_number_1 VARCHAR, 
	ifsc_code_1 VARCHAR, 
	bank_name_2 VARCHAR, 
	bank_account_number_2 VARCHAR, 
	ifsc_code_2 VARCHAR, 
	PRIMARY KEY (id), 
	UNIQUE (seller_id)
);

	shipping_gst_number VARCHAR, 
	broker_name VARCHAR,

	bank_name_2 VARCHAR, 
	bank_account_number_2 VARCHAR, 
	ifsc_code_2 VARCHAR,
